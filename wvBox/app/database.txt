





USE help_book_release

CREATE TABLE IF NOT EXISTS `twitters` (
    `id` int(20) NOT NULL AUTO_INCREMENT,
    `oauth_id` int(30),
    `username` varchar(255) NOT NULL,
    `showname` varchar(255) NOT NULL,
    `oauth_token` varchar(128) DEFAULT NULL,
    `oauth_token_secret` varchar(128) DEFAULT NULL,
    `created` datetime DEFAULT NULL,
    `modified` datetime DEFAULT NULL,
    PRIMARY KEY (`id`)
) ENGINE=InnoDB  DEFAULT CHARSET=utf8 ;


CREATE TABLE IF NOT EXISTS `topics` (
    `uuid` varchar(255) NOT NULL,
    `description` varchar(1048576) NOT NULL,
    `category` varchar(128) DEFAULT NOT NULL,
    `command` varchar(1048576) DEFAULT NULL,
    `configure` varchar(1048576) DEFAULT NULL,
    `created` datetime DEFAULT NULL,
    `modified` datetime DEFAULT NULL,
    PRIMARY KEY (`uuid`)
) ENGINE=InnoDB  DEFAULT CHARSET=utf8 ;
